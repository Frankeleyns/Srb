# 01-管理平台放款



## 一、前端整合

**srb-admin** 项目



### 1. Api

```js
  makeLoan(id){
    return request({
      url: `/admin/core/lend/makeLoan/${id}`,
      method: "get"
    })
  },
```



### 2. 脚本

```js
    // 放款
    makeLoan(id) {
      this.$confirm("确定放款吗？", "提示", {
        confirmButtonText: "确定",
        cancelButtonText: "取消",
        type: "warning",
      }).then( () => {
        lendApi.makeLoan(id)
      }).then(res => {
        // 处理放款结果
        this.fetchData();
        this.$message.success(res.message)
      }).catch(error => {
        if(error == 'cancel') {
          this.$message.info('已取消放款')
        }
      })
    },
```
